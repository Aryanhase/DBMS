CREATE TABLE N_RollCall (
	roll_no NUMBER PRIMARY KEY,
	name VARCHAR2(50)
);

CREATE TABLE O_RollCall (
	roll_no NUMBER PRIMARY KEY,
	name VARCHAR2(50)
);


--parameterized cursor
DECLARE                                      
	CURSOR new_data_cursor(p_roll_no NUMBER) IS
		SELECT roll_no, name
		FROM N_RollCall
		WHERE roll_no = p_roll_no;

	v_roll_no NUMBER;
	v_name VARCHAR2(50);
	v_exists NUMBER := 0;
BEGIN
	FOR new_record IN (SELECT roll_no, name FROM N_RollCall) LOOP
	
		OPEN new_data_cursor(new_record.roll_no);
		FETCH new_data_cursor INTO v_roll_no,v_name;

		SELECT COUNT(*) INTO v_exists
		FROM O_RollCall
		WHERE roll_no = new_record.roll_no;
		

		IF v_exists=0 THEN 
			INSERT INTO O_RollCall (roll_no, name)
			VALUES (new_record.roll_no, new_record.name);
		END IF;
		CLOSE new_data_cursor;
	END LOOP;
	COMMIT;
END;
